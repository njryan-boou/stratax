41inline ResolvedSlice normalize_slice(
42 const stratax::core::Slice& slice,
43 std::size_t extent,
44 const char* message)
45{
46 if (extent > static_cast<std::size_t>(std::numeric_limits<std::ptrdiff_t>::max()))
47 {
48 throw Exceptions::IndexError(message);
49 }
50
51 const std::ptrdiff_t n = static_cast<std::ptrdiff_t>(extent);
52 std::ptrdiff_t start = slice.start();
53 std::ptrdiff_t stop = slice.stop();
54 const std::ptrdiff_t step = slice.step();
55
56 if (start < 0)
57 {
58 start += n;
59 }
60 if (stop < 0)
61 {
62 stop += n;
63 }
64
65 if (step > 0)
66 {
67 start = clamp(start, 0, n);
68 stop = clamp(stop, 0, n);
69
70 if (start >= stop)
71 {
72 return ResolvedSlice{start, step, 0};
73 }
74
75 const std::ptrdiff_t distance = stop - start;
76 const std::size_t count = static_cast<std::size_t>((distance + step - 1) / step);
77 return ResolvedSlice{start, step, count};
78 }
79
80 start = clamp(start, -1, n - 1);
81 stop = clamp(stop, -1, n - 1);
82
83 if (start <= stop)
84 {
85 return ResolvedSlice{start, step, 0};
86 }
87
88 const std::ptrdiff_t stride = -step;
89 const std::ptrdiff_t distance = start - stop;
90 const std::size_t count = static_cast<std::size_t>((distance + stride - 1) / stride);
91 return ResolvedSlice{start, step, count};
92}

231stratax::container::Tensor<T>
232slice(
233 const stratax::container::Tensor<T>& tensor,
234 Slices... slices
235)
236{
237 static_assert(

239 "All arguments must be Slice."
240 );
241
242 std::array<stratax::core::Slice, sizeof...(Slices)> ranges{ slices... };
243
244 stratax::core::validation::require_rank(
245 ranges.size(),
246 tensor.rank(),
247 "Slice rank must match tensor rank.");
248
249 std::array<stratax::ops::detail::ResolvedSlice, sizeof...(Slices)> resolved{};
250 std::array<std::size_t, sizeof...(Slices)> out_dims{};
251 for (std::size_t dim = 0; dim < ranges.size(); ++dim)
252 {
253 resolved[dim] = stratax::ops::detail::normalize_slice(
254 ranges[dim],
255 tensor.shape()(dim),
256 "Tensor slice out of bounds.");
257 out_dims[dim] = resolved[dim].size;
258 }
259
260 const auto result_shape = stratax::core::Shape(
261 std::vector<std::size_t>(out_dims.begin(), out_dims.end()));
262 stratax::container::Tensor<T> result(result_shape);
263 const stratax::core::Strides result_strides(result_shape);
264 const auto& tensor_strides = tensor.strides();
265
266 if (result.empty())
267 {
268 return result;
269 }
270
271 for (std::size_t flat = 0; flat < result.size(); ++flat)
272 {
273 std::size_t remainder = flat;
274 std::size_t source_offset = 0;
275
276 for (std::size_t dim = 0; dim < resolved.size(); ++dim)
277 {
278 const std::size_t index = remainder / result_strides(dim);
279 remainder %= result_strides(dim);
280
281 const std::ptrdiff_t source_index =
282 resolved[dim].start + static_cast<std::ptrdiff_t>(index) * resolved[dim].step;
283 const std::size_t term =
284 stratax::core::validation::checked_multiply(
285 static_cast<std::size_t>(source_index),
286 tensor_strides(dim),
287 "Tensor slice offset overflow.");
288 source_offset =
289 stratax::core::validation::checked_add(
290 source_offset,
