104 Matrix(std::initializer_list<std::initializer_list<T>> list)
105 {
106 std::size_t rows = list.size();
107 std::size_t cols = (rows == 0) ? 0 : list.begin()->size();
108
109 // Ensure all rows have the same length
110 for (const auto& row : list)
111 {
112 if (row.size() != cols)
113 {
114 throw Exceptions::ShapeError("Matrix initializer rows must all have the same number of columns.");
115 }
116 }
117
118 shape_ = stratax::core::Shape({rows, cols}, stratax::core::Shape::allow_zero);
119 strides_ = stratax::core::Strides(shape_);
120 buffer_ = stratax::core::Buffer<T>(
121 core::validation::checked_multiply(rows, cols, "Matrix size overflow"));
122
123 std::size_t index = 0;
124
125 for (const auto& row : list)
126 {
127 for (const auto& value : row)
128 {
129 buffer_[index++] = value;
130 }
131 }
